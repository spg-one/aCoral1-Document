203/*
204 * Macros to set/get and increase/decrease variables in a thread-safe way.
205 * Use these for accessing variable that are used from more than one thread.
206 */
207
208#ifndef SYS_ARCH_INC
209#define SYS_ARCH_INC(var, val) do { \
210 SYS_ARCH_DECL_PROTECT(old_level); \
211 SYS_ARCH_PROTECT(old_level); \
212 var += val; \
213 SYS_ARCH_UNPROTECT(old_level); \
214 } while(0)
215#endif /* SYS_ARCH_INC */
216
217#ifndef SYS_ARCH_DEC
218#define SYS_ARCH_DEC(var, val) do { \
219 SYS_ARCH_DECL_PROTECT(old_level); \
220 SYS_ARCH_PROTECT(old_level); \
221 var -= val; \
222 SYS_ARCH_UNPROTECT(old_level); \
223 } while(0)
224#endif /* SYS_ARCH_DEC */
225
226#ifndef SYS_ARCH_GET
227#define SYS_ARCH_GET(var, ret) do { \
228 SYS_ARCH_DECL_PROTECT(old_level); \
229 SYS_ARCH_PROTECT(old_level); \
230 ret = var; \
231 SYS_ARCH_UNPROTECT(old_level); \
232 } while(0)
